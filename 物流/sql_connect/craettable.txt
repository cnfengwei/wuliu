CREATE TABLE `jd_bill` (
	`waybill` CHAR(20) NOT NULL DEFAULT '' COLLATE 'utf8mb4_general_ci',
	`start_time` DATETIME NOT NULL,
	`end_time` DATETIME NOT NULL,
	`thirdbill` CHAR(11) NOT NULL COLLATE 'utf8mb4_general_ci',
	`start_point` CHAR(20) NOT NULL COLLATE 'utf8mb4_general_ci',
	`end_point` CHAR(20) NOT NULL COLLATE 'utf8mb4_general_ci',
	` vehicle_num` CHAR(20) NOT NULL COLLATE 'utf8mb4_general_ci',
	` vehicle_type` CHAR(20) NOT NULL COLLATE 'utf8mb4_general_ci',
	`third_driver` CHAR(4) NOT NULL COLLATE 'utf8mb4_general_ci',
	`drivers` CHAR(4) NOT NULL COLLATE 'utf8mb4_general_ci',
	PRIMARY KEY (`waybill`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
;
